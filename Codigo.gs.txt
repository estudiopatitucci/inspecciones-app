// ─── SEGURIDAD — validación de entrada y rate limiting del endpoint ───────────

// Coerce a string, recorta y limita longitud (anti payloads gigantes / basura).
function _str(v, max) {
  var s = (v == null ? '' : String(v)).trim();
  return max ? s.slice(0, max) : s;
}

// Rate limit simple por clave usando CacheService. Devuelve true si está permitido.
function _rateLimit(key, maxReqs, windowSec) {
  try {
    var cache = CacheService.getScriptCache();
    var k = 'rl_' + key;
    var n = Number(cache.get(k) || 0) + 1;
    cache.put(k, String(n), windowSec);
    return n <= maxReqs;
  } catch (e) {
    return true; // si el cache falla, no bloqueamos al usuario legítimo
  }
}

// Validación básica de formato de email.
function _emailValido(email) {
  return /^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(email);
}

// ─── CONCURRENCIA — serializa escrituras al Sheet para evitar race conditions ──
// Ejecuta fn() tomando un lock global. Si el sistema está ocupado, avisa al
// usuario en vez de pisar datos. Libera siempre el lock al terminar.
function _withLock(fn) {
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(20000); // espera hasta 20 s a que se libere
  } catch (e) {
    return resp({ok: false, msg: 'El sistema está procesando otra operación. Probá de nuevo en unos segundos.'});
  }
  try {
    return fn();
  } finally {
    try { lock.releaseLock(); } catch (e) {}
  }
}

// Lee todos los templates de la Sheet. Devuelve objeto {key: {asunto, cuerpo}}
function _leerMailConfig() {
  var sh = _getMailConfigSheet();
  var rows = sh.getDataRange().getValues();
  var cfg = {};
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0]) cfg[rows[i][0]] = { asunto: rows[i][1] || '', cuerpo: rows[i][2] || '' };
  }
  return cfg;
}

// Obtiene asunto y cuerpo para una key, con fallback a MAIL_DEFAULTS
function _getTpl(key) {
  var cfg = _leerMailConfig();
  var def = MAIL_DEFAULTS[key] || { asunto: '', cuerpo: '' };
  return {
    asunto: (cfg[key] && cfg[key].asunto) ? cfg[key].asunto : def.asunto,
    cuerpo: (cfg[key] && cfg[key].cuerpo) ? cfg[key].cuerpo : def.cuerpo
  };
}

// Reemplaza variables {{var}} en un string
function _tplReplace(str, vars) {
  return str.replace(/\{\{(\w+)\}\}/g, function(_, k) {
    return (vars[k] !== undefined && vars[k] !== null) ? String(vars[k]) : '';
  });
}

// Endpoint: devuelve todos los templates (defaults + overrides de la Sheet)
function getMailTemplates() {
  var cfg = _leerMailConfig();
  var result = [];
  Object.keys(MAIL_DEFAULTS).forEach(function(key) {
    var def = MAIL_DEFAULTS[key];
    var override = cfg[key] || {};
    result.push({
      key: key,
      label: def.label,
      asunto: override.asunto || def.asunto,
      cuerpo: override.cuerpo || def.cuerpo,
      modificado: !!(override.asunto || override.cuerpo)
    });
  });
  return resp({ ok: true, templates: result });
}

// Endpoint: guarda un template en la Sheet
function saveMailTemplate(data) {
  if (data.usuario !== 'Administrador') return resp({ ok: false, msg: 'Sin permiso.' });
  var key    = _str(data.key, 60);
  var asunto = _str(data.asunto, 500);
  var cuerpo = _str(data.cuerpo, 5000);
  if (!key || !MAIL_DEFAULTS[key]) return resp({ ok: false, msg: 'Template inválido.' });
  var sh   = _getMailConfigSheet();
  var rows = sh.getDataRange().getValues();
  var ahora = new Date().toISOString();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] === key) {
      sh.getRange(i+1, 2).setValue(asunto);
      sh.getRange(i+1, 3).setValue(cuerpo);
      sh.getRange(i+1, 4).setValue(ahora);
      return resp({ ok: true });
    }
  }
  sh.appendRow([key, asunto, cuerpo, ahora]);
  return resp({ ok: true });
}

// Endpoint: restaura un template a su valor por defecto
function resetMailTemplate(data) {
  if (data.usuario !== 'Administrador') return resp({ ok: false, msg: 'Sin permiso.' });
  var key = _str(data.key, 60);
  if (!key || !MAIL_DEFAULTS[key]) return resp({ ok: false, msg: 'Template inválido.' });
  var sh   = _getMailConfigSheet();
  var rows = sh.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] === key) {
      sh.deleteRow(i+1);
      return resp({ ok: true });
    }
  }
  return resp({ ok: true }); // no existía override, ya estaba en default
}

function doPost(e) {

    if (!e || !e.postData || !e.postData.contents) {
      return resp({ok: false, msg: 'Solicitud inválida.'});
    }
    var data;
    try { data = JSON.parse(e.postData.contents); }
    catch (parseErr) { return resp({ok: false, msg: 'Solicitud malformada.'}); }
    if (!data || typeof data !== 'object') return resp({ok: false, msg: 'Solicitud inválida.'});

    var action = data.action;
    // relevamientoTaller usa tallerToken propio, no SECRET_TOKEN. Escribe → lock.
    if (action === 'relevamientoTaller') return _withLock(function(){ return recibirRelTaller(data); });
    if (action === 'uploadFotoTaller') return uploadFotoTaller(data);
    // Acciones del taller vía verifTaller.html — autenticadas por verifToken, no SECRET_TOKEN

      return resp({ok: false, msg: 'No autorizado'});
    }
    if (action === 'login') return loginUsuario(data);
    // Acciones que escriben en el Sheet → serializadas con _withLock
    if (action === 'cambiarClave') return _withLock(function(){ return cambiarClave(data); });
    if (action === 'guardar') return _withLock(function(){ return guardarInspeccion(data); });
    if (action === 'actualizar') return _withLock(function(){ return actualizarInspeccion(data); });

  var token = e && e.parameter && e.parameter.token;
  if (token !== SECRET_TOKEN) return resp({ok: false, msg: 'No autorizado'});
  if (action === 'getArchivadas') return getArchivadas();
  if (action === 'getFile') return getFilePorId(e.parameter.fileId);
  return listarInspecciones();
}

function getFilePorId(fileId) {
  if (!fileId) return resp({ok: false, msg: 'fileId requerido'});
  try {
    var file = DriveApp.getFileById(fileId);
    var blob = file.getBlob();
    var bytes = blob.getBytes();
    var base64 = Utilities.base64Encode(bytes);
    return resp({ok: true, base64: base64, mimeType: blob.getContentType() || 'image/jpeg', name: file.getName()});
  } catch(e) {
    return resp({ok: false, msg: e.message});
  }
}

// Devuelve el número de columna (1-based) de DanosDen, creándola si no existe.
function getDanosDenCol(sheet) {
  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  var idx = headers.indexOf('DanosDen');
  if (idx !== -1) return idx + 1;
  // No existe → agregar al final
  var col = sheet.getLastColumn() + 1;
  sheet.getRange(1, col).setValue('DanosDen');
  return col;
}

function leerInspecciones(sheet) {
  var rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) return [];
  var headers = rows[0];
  var result = [];
  for (var i = 1; i < rows.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = rows[i][j];
    }
    if (obj.Relevamiento) {
      try { obj.Relevamiento = JSON.parse(obj.Relevamiento); }
      catch(e) { obj.Relevamiento = null; }
    }
    if (obj.Devoluciones) {
      try { obj.Devoluciones = JSON.parse(obj.Devoluciones); }
      catch(e) { obj.Devoluciones = []; }
    }
    result.push(obj);
  }
  return result;
}

// Nombre de la carpeta madre donde se guardan todas las inspecciones
var CARPETA_MADRE = 'Inspecciones Patitucci - Fotos';

function getCarpetaMadre() {
  var folders = DriveApp.getFoldersByName(CARPETA_MADRE);
  if (folders.hasNext()) return folders.next();
  // Si no existe, la crea y comparte con los viewers autorizados
  var nueva = DriveApp.createFolder(CARPETA_MADRE);
  try { compartirConAutorizados(nueva); } catch(e) {}
  return nueva;
}

function getOrCreateFolder(name) {
  var madre = getCarpetaMadre();
  // Buscar la subcarpeta dentro de la carpeta madre
  var subs = madre.getFoldersByName(name);
  if (subs.hasNext()) return subs.next();
  // Si no existe dentro de la madre, buscar en el raíz (carpetas viejas sueltas) y moverlas
  var suelta = DriveApp.getFoldersByName(name);
  if (suelta.hasNext()) {
    var carpetaSuelta = suelta.next();
    madre.addFolder(carpetaSuelta);
    DriveApp.getRootFolder().removeFolder(carpetaSuelta);
    return carpetaSuelta;
  }
  // Crear dentro de la carpeta madre
  return madre.createFolder(name);
}

function actualizarInspeccion(data) {
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.id) {
      if (data.estado !== undefined) sheet.getRange(i+1, 20).setValue(data.estado);
      if (data.relevamiento !== undefined) {
        var rel = data.relevamiento;
        // Si hay fotos en base64, subirlas a Drive
        // Flujo nuevo: fotos ya subidas progresivamente desde el frontend
        if (rel.fotosLinks && rel.fotosLinks.length > 0 && (!rel.fotos || rel.fotos.length === 0)) {
          rel.cantFotos = rel.fotosLinks.length;
          try {
            var folder = getOrCreateFolder('Inspeccion_' + data.id);
            rel.carpetaUrl = folder.getUrl();
          } catch(eF) {}
        }
        // Flujo legacy: fotos en base64 (compatibilidad con versiones anteriores)
        if (rel.fotos && rel.fotos.length > 0) {
          try {
            var folder = getOrCreateFolder('Inspeccion_' + data.id);
            // Intentar restringir la carpeta — fallo no debe bloquear la subida
            try { compartirConAutorizados(folder); } catch(eFolder) {}
            var links = [];
            rel.fotos.forEach(function(base64, idx) {
              try {
                var b64 = base64.replace(/^data:image\/\w+;base64,/, '');
                var blob = Utilities.newBlob(Utilities.base64Decode(b64), 'image/jpeg', 'foto_' + (idx+1) + '.jpg');
                var file = folder.createFile(blob);
                // Hacer cada foto accesible con el link directo (no indexable, no buscable)
                // Esto permite que el panel admin muestre miniaturas sin depender de la cuenta Google del navegador
                try { file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(eShare) {}
                // También agregar como viewer a los usuarios autorizados
                try { compartirConAutorizados(file); } catch(eFile) {}
                // Guardar link y thumbUrl listo para usar en el frontend
                var fileId = file.getId();
                links.push({
                  viewUrl: file.getUrl(),
                  id: fileId,
                  thumbUrl: 'https://lh3.googleusercontent.com/d/' + fileId + '=w400'
                });
              } catch(e) {}
            });
            rel.fotosLinks = links;
            rel.cantFotos = links.length;
            rel.carpetaUrl = folder.getUrl();
          } catch(e) {}
          delete rel.fotos; // Eliminar base64 antes de guardar en Sheets
        }
        sheet.getRange(i+1, 22).setValue(JSON.stringify(rel));
        // Registrar o actualizar taller en la base de talleres
        try { registrarTaller({nombre:rel.taller, mail:rel.mailTaller, tel:rel.telTaller, cuit:rel.cuit, dir:rel.dirTaller, loc:rel.locTaller}); } catch(eT) {}
        // Notificar al estudio que el inspector cargo el relevamiento
        try {

function archivarInspecciones(data) {
  if (!data || data.usuario !== 'Administrador') return resp({ok: false, msg: 'Sin permiso.'});
  var sheet = getSheet();
  var archivoSheet = getArchivoSheet();
  var rows = sheet.getDataRange().getValues();
  var headers = rows[0];
  var hoy = new Date();
  var seisM = new Date(hoy);
  seisM.setMonth(seisM.getMonth() - 6);

  var archivadas = 0;
  // Recorrer de abajo para arriba para no alterar índices al borrar
  for (var i = rows.length - 1; i >= 1; i--) {
    var estado = rows[i][19]; // columna Estado (índice 19)
    var fechaInsp = rows[i][4]; // columna FechaInsp (índice 4)
    if (estado !== 'Cerradas') continue;
    var fecha = new Date(fechaInsp);
    if (isNaN(fecha.getTime())) continue;
    if (fecha > seisM) continue;
    // Mover a Archivo
    var fila = rows[i].slice();
    fila.push(new Date().toISOString()); // ArchivedAt
    archivoSheet.appendRow(fila);
    sheet.deleteRow(i + 1);
    archivadas++;
  }
  return {ok: true, archivadas: archivadas};
}

function enviarContacto(data) {
  try {
    var nombre   = _str(data.nombre, 120);
    var email    = _str(data.email, 200);
    var mensaje  = _str(data.mensaje, 5000);
    var empresa  = _str(data.empresa, 150);
    var telefono = _str(data.telefono, 50);

    if (!nombre || !email || !mensaje) return resp({ok: false, msg: 'Completá nombre, email y mensaje.'});
    if (!_emailValido(email)) return resp({ok: false, msg: 'Email inválido.'});

    // Anti spam: máx 3 envíos por email cada 10 minutos
    if (!_rateLimit('contacto_' + email.toLowerCase(), 3, 600)) {
      return resp({ok: false, msg: 'Recibimos tu consulta hace poco. Aguardá unos minutos antes de reenviar.'});
    }

    var asunto = 'Consulta desde estudiopatitucci.com.ar — ' + nombre;
    var cuerpo = 'Nueva consulta desde el sitio web:\n\n'
      + 'Nombre: ' + nombre + '\n'
      + 'Email: ' + email + '\n'
      + (empresa ? 'Empresa: ' + empresa + '\n' : '')
      + (telefono ? 'Teléfono: ' + telefono + '\n' : '')
      + '\nMensaje:\n' + mensaje;
    MailApp.sendEmail({
      to: 'patitucciestudio@gmail.com',
      replyTo: email,
      subject: asunto,
      body: cuerpo
    });
    return resp({ok: true});
  } catch(err) {
    Logger.log('enviarContacto error: ' + err);
    return resp({ok: false, msg: 'No se pudo enviar la consulta. Intentá de nuevo.'});
  }
}

//  Correr UNA SOLA VEZ después de hacer deploy
//  Pasos: seleccionar "migrarCarpetasAMadre" → Ejecutar → Ver Registros
// ══════════════════════════════════════════════════════
function migrarCarpetasAMadre() {
  var madre = getCarpetaMadre();
  var query = DriveApp.searchFolders('title contains "Inspeccion_"');
  var movidas = 0, ya = 0;
  while (query.hasNext()) {
    var carpeta = query.next();
    var parents = carpeta.getParents();
    var estaEnMadre = false;
    while (parents.hasNext()) {
      if (parents.next().getId() === madre.getId()) { estaEnMadre = true; break; }
    }
    if (estaEnMadre) { ya++; continue; }
    madre.addFolder(carpeta);
    DriveApp.getRootFolder().removeFolder(carpeta);
    movidas++;
    Logger.log('Movida: ' + carpeta.getName());
  }
  Logger.log('Ya estaban en madre: ' + ya + ' | Movidas ahora: ' + movidas);
}

// ================================================================
//  FEATURE: LINK PARA TALLERES
//  El admin genera un link unico con vencimiento de 48 hs.
//  El taller lo abre, carga el relevamiento y lo envia.
//  No requiere login. El token se invalida al usarse o al vencer.
// ================================================================

function generarToken32() {
  var chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  var t = '';
  for (var i = 0; i < 32; i++) t += chars[Math.floor(Math.random() * chars.length)];
  return t;
}

function generarTokenTaller(data) {
  if (!data.id) return resp({ok: false, msg: 'ID requerido'});
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.id) {
      var token = generarToken32();
      var exp = new Date();
      exp.setHours(exp.getHours() + 48);
      sheet.getRange(i+1, 25).setValue(token);        // col TallerToken
      sheet.getRange(i+1, 26).setValue(exp.toISOString()); // col TallerTokenExp
      sheet.getRange(i+1, 20).setValue('Esperando taller'); // col Estado
      var appUrl = 'https://estudiopatitucci.github.io/inspecciones-app/?taller=' + token;
      return resp({ok: true, link: appUrl, exp: exp.toISOString()});
    }
  }
  return resp({ok: false, msg: 'Inspeccion no encontrada'});
}

      if (rel.fotosLinks && rel.fotosLinks.length > 0 && (!rel.fotos || rel.fotos.length === 0)) {

function getTalleresSheet() {
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var sheet = ss.getSheetByName(TALLERES_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(TALLERES_NAME);
    sheet.appendRow(['ID','Nombre','Mail','Dirección','Localidad','Teléfono','CUIT','FechaAlta','CantUsos']);
    sheet.getRange(1,1,1,9).setFontWeight('bold');
  }
  return sheet;
}

// Normaliza strings para comparación tolerante (sin tildes, minúsculas)
function _normTaller(str) {
  return String(str||'').toLowerCase()
    .replace(/[áàâä]/g,'a').replace(/[éèêë]/g,'e')
    .replace(/[íìîï]/g,'i').replace(/[óòôö]/g,'o')
    .replace(/[úùûü]/g,'u').replace(/ñ/g,'n')
    .replace(/[^a-z0-9\s]/g,'').replace(/\s+/g,' ').trim();
}

// Busca talleres por nombre (mínimo 2 caracteres). Retorna hasta 6 matches ordenados por CantUsos.
function buscarTalleres(data) {
  if (!data.query || data.query.trim().length < 3) return resp({ok:true, talleres:[]});
  var sheet = getTalleresSheet();
  var rows = sheet.getDataRange().getValues();
  var q = _normTaller(data.query);
  var matches = [];
  for (var i = 1; i < rows.length; i++) {
    if (_normTaller(String(rows[i][1]||'')).indexOf(q) !== -1) {
      matches.push({
        nombre:   String(rows[i][1]||''),
        mail:     String(rows[i][2]||''),
        dir:      String(rows[i][3]||''),
        loc:      String(rows[i][4]||''),
        tel:      String(rows[i][5]||''),
        cuit:     String(rows[i][6]||''),
        cantUsos: Number(rows[i][8]||0)
      });
    }
  }
  matches.sort(function(a,b){ return b.cantUsos - a.cantUsos; });
  return resp({ok:true, talleres: matches.slice(0,6)});
}

// Agrega el taller si no existe, o actualiza campos vacíos y suma un uso.
// Deduplicación: CUIT (prioritario) → Nombre+Mail (fallback).
function registrarTaller(t) {
  if (!t || !t.nombre) return;
  try {
    var sheet = getTalleresSheet();
    var rows = sheet.getDataRange().getValues();
    var cuit = String(t.cuit||'').replace(/[^0-9]/g,'');
    // Buscar por CUIT  (col 7 = índice 6)
    if (cuit && cuit.length === 11) {
      for (var i = 1; i < rows.length; i++) {
        if (String(rows[i][6]||'').replace(/[^0-9]/g,'') === cuit) {
          if (!rows[i][2] && t.mail) sheet.getRange(i+1,3).setValue(t.mail);
          if (!rows[i][3] && t.dir)  sheet.getRange(i+1,4).setValue(t.dir);
          if (!rows[i][4] && t.loc)  sheet.getRange(i+1,5).setValue(t.loc);
          if (!rows[i][5] && t.tel)  sheet.getRange(i+1,6).setValue(t.tel);
          sheet.getRange(i+1,9).setValue((Number(rows[i][8])||0)+1);
          return;
        }
      }
    }
    // Fallback: Nombre + Mail
    var nn = _normTaller(t.nombre);
    var nm = String(t.mail||'').toLowerCase().trim();
    for (var j = 1; j < rows.length; j++) {
      if (_normTaller(String(rows[j][1]||'')) === nn &&
          String(rows[j][2]||'').toLowerCase().trim() === nm) {
        if (!rows[j][3] && t.dir)  sheet.getRange(j+1,4).setValue(t.dir);
        if (!rows[j][4] && t.loc)  sheet.getRange(j+1,5).setValue(t.loc);
        if (!rows[j][5] && t.tel)  sheet.getRange(j+1,6).setValue(t.tel);
        if (!rows[j][6] && cuit)   sheet.getRange(j+1,7).setValue(cuit);
        sheet.getRange(j+1,9).setValue((Number(rows[j][8])||0)+1);
        return;
      }
    }
    // Nuevo taller
    sheet.appendRow([
      Date.now().toString(),
      t.nombre,
      t.mail  || '',
      t.dir   || '',
      t.loc   || '',
      t.tel   || '',
      cuit    || '',
      new Date().toISOString(),
      1
    ]);
  } catch(e) {
    Logger.log('registrarTaller error: ' + e);
  }
}

function analizarPresupuesto(data) {
  try {
    var apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_KEY');
    if (!apiKey) {
      return resp({ok: false, msg: 'API key de Gemini no configurada. Ir a GAS Editor → Proyecto → Configuración → Propiedades de secuencia de comandos → agregar GEMINI_KEY con tu key de aistudio.google.com'});
    }
    if (!data.fileBase64 || !data.mimeType) {
      return resp({ok: false, msg: 'Se requiere fileBase64 y mimeType'});
    }

    // Quitar el prefijo "data:...;base64," si viene del frontend
    var base64 = data.fileBase64.replace(/^data:[^;]+;base64,/, '');

    // Llamar a Gemini Vision
    var geminiResult = callGeminiAPI(base64, data.mimeType);
    if (!geminiResult.ok) return resp({ok: false, msg: geminiResult.error});

    // Parsear la respuesta JSON de Gemini
    var piezas = parsearRespuestaGemini(geminiResult.text);
    if (!piezas || piezas.length === 0) {
      return resp({ok: false, msg: 'Gemini no encontró piezas en el documento. Verificá que la imagen sea un presupuesto de taller legible.'});
    }

    // Mapear a zonas y piezas del calculador
    var resultado = mapPiezasToCalc(piezas);
    return resp({ok: true, mapeadas: resultado.mapeadas, sinMapear: resultado.sinMapear});

  } catch(err) {
    return resp({ok: false, msg: 'Error al analizar presupuesto: ' + err.toString()});
  }
}

        { inlineData: { mimeType: mimeType, data: base64 } },
        { text: prompt }
      ]
    }],
    generationConfig: {
      temperature: 0.1,
      maxOutputTokens: 2048,
      thinkingConfig: { thinkingBudget: 0 }
    }
  };

  var options = {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  var response = UrlFetchApp.fetch(url, options);
  var code = response.getResponseCode();
  var body;
  try { body = JSON.parse(response.getContentText()); }
  catch(e) { return {ok: false, error: 'Respuesta inválida de Gemini (HTTP ' + code + ')'}; }

  if (code !== 200) {
    var errMsg = (body.error && body.error.message) ? body.error.message : 'Error HTTP ' + code;
    return {ok: false, error: 'Gemini API: ' + errMsg};
  }

  try {
    var text = body.candidates[0].content.parts[0].text;
    return {ok: true, text: text};
  } catch(e) {
    return {ok: false, error: 'No se pudo leer la respuesta de Gemini'};
  }
}

// ── Parsear el JSON devuelto por Gemini ───────────────────
function parsearRespuestaGemini(text) {
  try {
    // Limpiar posibles restos de markdown que Gemini a veces agrega igual
    var clean = text.trim()
      .replace(new RegExp('^```json\\s*', 'i'), '')
      .replace(new RegExp('^```\\s*', 'i'), '')
      .replace(new RegExp('\\s*```$', 'i'), '');

    var arr = JSON.parse(clean);
    if (!Array.isArray(arr)) return null;

    return arr
      .map(function(item) {
        return {
          pieza:         (item.pieza || '').trim(),
          reparar:       item.reparar === true,
          sustituir:     item.sustituir === true,
          pintura:       item.pintura === true,
          desabolladura: item.desabolladura === true
        };
      })
      .filter(function(item) {
        return item.pieza && (item.reparar || item.sustituir || item.pintura || item.desabolladura);
      });

  } catch(e) {
    return null;
  }
}

// ── Normalizar texto para matching (sin tildes, minúsculas) ─
function normalizarPieza(str) {
  return String(str || '').toLowerCase()
    .replace(/[áàâä]/g,'a').replace(/[éèêë]/g,'e')
    .replace(/[íìîï]/g,'i').replace(/[óòôö]/g,'o')
    .replace(/[úùûü]/g,'u').replace(/ñ/g,'n')
    .replace(/[^a-z0-9\s\-\/]/g,'')
    .replace(/\s+/g,' ').trim();
}

// ── Mapear piezas de Gemini → zonas y piezas del calculador ─
// Cada pieza con múltiples tareas genera una fila por tarea,
// porque así trabaja relCalcState.piezas en el frontend.
// Nivel de daño para Reparar y Desabolladura: siempre 'leve'
// (el admin lo ajusta en el Panel de Revisión si corresponde).
function mapPiezasToCalc(piezas) {
  var mapeadas = [];
  var sinMapear = [];

  piezas.forEach(function(item) {
    var keyNorm = normalizarPieza(item.pieza);
    var match = PIEZAS_MAP[keyNorm] || null;

    // Si no hay match exacto, intentar match parcial buscando la clave que más se parezca
    if (!match) {
      var mejorScore = 0;
      var mejorKey = null;
      Object.keys(PIEZAS_MAP).forEach(function(k) {
        // Score simple: cuántas palabras de la clave están en el texto buscado
        var wordsK = k.split(' ');
        var wordsN = keyNorm.split(' ');
        var comunes = wordsK.filter(function(w){ return w.length > 3 && wordsN.indexOf(w) !== -1; });
        var score = comunes.length / Math.max(wordsK.length, 1);
        if (score > mejorScore) { mejorScore = score; mejorKey = k; }
      });
      if (mejorScore >= 0.6) match = PIEZAS_MAP[mejorKey];
    }

    // Generar una fila por cada tarea que tenga la pieza
    var tareas = [];
    if (item.sustituir)     tareas.push({tarea:'Sustituir',  nivelDano: null});
    if (item.reparar)       tareas.push({tarea:'Reparar',    nivelDano: 'leve'});
    if (item.desabolladura) tareas.push({tarea:'Reparar',    nivelDano: 'leve'});  // desabolladura → Reparar leve
    if (item.pintura)       tareas.push({tarea:'Pintar',     nivelDano: null});

    // Deduplicar si reparar+desabolladura generaron dos filas iguales
    var tareasSeen = {};
    tareas = tareas.filter(function(t) {
      var k = t.tarea + (t.nivelDano || '');
      if (tareasSeen[k]) return false;
      tareasSeen[k] = true;
      return true;
    });

    tareas.forEach(function(t) {
      if (match) {
        mapeadas.push({
          piezaOriginal: item.pieza,
          zona:          match.zona,
          keyword:       match.keyword,  // el frontend lo usa para buscar en CALC_DATA_GRID
          tarea:         t.tarea,
          nivelDano:     t.nivelDano
        });
      } else {
        sinMapear.push({
          piezaOriginal: item.pieza,
          tarea:         t.tarea,
          nivelDano:     t.nivelDano
        });
      }
    });
  });

  return {mapeadas: mapeadas, sinMapear: sinMapear};
}

// ─── Hash de contraseñas (SHA-256 + salt) ────────────────────────────────────
// Formato almacenado en la columna Clave: "v1$<saltHex>$<hashHex>"
// Las claves en texto plano (legacy) se siguen aceptando y se re-hashean al usarse.
var HASH_PREFIX = 'v1$';

function _generarSalt() {
  // 32 caracteres hex (UUID sin guiones) — único por usuario
  return Utilities.getUuid().replace(/-/g, '');
}

function _hashClave(clave, saltHex) {
  var raw = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    saltHex + '|' + clave,
    Utilities.Charset.UTF_8
  );
  var hex = '';
  for (var i = 0; i < raw.length; i++) {
    hex += ('0' + (raw[i] & 0xFF).toString(16)).slice(-2);
  }
  return hex;
}

function _formatoHash(clave) {
  var salt = _generarSalt();
  return HASH_PREFIX + salt + '$' + _hashClave(clave, salt);
}

// Devuelve true si la clave coincide con lo almacenado (hasheado o legacy en plano).
function _verificarClave(clave, almacenado) {
  almacenado = String(almacenado);
  if (almacenado.indexOf(HASH_PREFIX) === 0) {
    var parts = almacenado.split('$'); // ['v1', salt, hash]
    if (parts.length !== 3) return false;
    return _hashClave(clave, parts[1]) === parts[2];
  }
  // Legacy: comparación directa con texto plano
  return almacenado === clave;
}

function _esHasheada(almacenado) {
  return String(almacenado).indexOf(HASH_PREFIX) === 0;
}

function loginUsuario(data) {
  var nombre = (data.nombre || '').trim();
  var clave  = (data.clave  || '').trim();
  if (!nombre || !clave) return resp({ok: false, msg: 'Datos incompletos.'});

  // Anti fuerza bruta: máx 10 intentos por usuario cada 5 minutos
  if (!_rateLimit('login_' + nombre.toLowerCase(), 10, 300)) {
    return resp({ok: false, msg: 'Demasiados intentos. Esperá unos minutos e intentá de nuevo.'});
  }

  var sheet = getUsuariosSheet();
  var rows  = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] === nombre) {
      if (_verificarClave(clave, rows[i][1])) {
        // Migración lazy: si entró con una clave en texto plano, la re-guardamos hasheada
        if (!_esHasheada(rows[i][1])) {
          try { sheet.getRange(i + 1, 2).setValue(_formatoHash(clave)); } catch (e) {}
        }
        return resp({ok: true, nombre: nombre});
      } else {
        return resp({ok: false, msg: 'Contraseña incorrecta.'});
      }
    }
  }
  return resp({ok: false, msg: 'Usuario no encontrado.'});
}

// Copia la planilla a una carpeta aparte en Drive todos los días y borra las
// copias más viejas que BACKUP_RETENCION_DIAS. Red de seguridad ante borrados.

// Devuelve (creándola si hace falta) la carpeta donde se guardan los backups.
function _getCarpetaBackups() {
  var it = DriveApp.getFoldersByName(BACKUP_FOLDER_NAME);
  if (it.hasNext()) return it.next();
  var carpeta = DriveApp.createFolder(BACKUP_FOLDER_NAME);
  try { compartirConAutorizados(carpeta); } catch (e) {}
  return carpeta;
}

// Función que ejecuta el trigger diario. También se puede correr a mano.
function backupDiario() {
  var carpeta = _getCarpetaBackups();
  var tz = Session.getScriptTimeZone() || 'America/Argentina/Buenos_Aires';
  var sello = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd_HH-mm');
  var nombre = 'Backup Inspecciones — ' + sello;

  var copia = DriveApp.getFileById(SHEET_ID).makeCopy(nombre, carpeta);
  try { compartirConAutorizados(copia); } catch (e) {}

  _limpiarBackupsViejos(carpeta);
  Logger.log('Backup creado: ' + nombre + ' (' + copia.getId() + ')');
  return copia.getId();
}

// Borra de la carpeta de backups las copias más viejas que BACKUP_RETENCION_DIAS.
function _limpiarBackupsViejos(carpeta) {
  carpeta = carpeta || _getCarpetaBackups();
  var limite = new Date().getTime() - BACKUP_RETENCION_DIAS * 24 * 60 * 60 * 1000;
  var archivos = carpeta.getFiles();
  var borrados = 0;
  while (archivos.hasNext()) {
    var f = archivos.next();
    if (f.getName().indexOf('Backup Inspecciones') === 0 &&
        f.getDateCreated().getTime() < limite) {
      f.setTrashed(true); // a la papelera, recuperable 30 días
      borrados++;
    }
  }
  Logger.log('Backups viejos enviados a papelera: ' + borrados);
  return borrados;
}

// Instala el trigger diario (corre 1 vez desde el editor). Evita duplicados.
function instalarTriggerBackup() {
  eliminarTriggerBackup();
  ScriptApp.newTrigger('backupDiario')
    .timeBased()
    .everyDays(1)
    .atHour(3) // ~3 AM hora del script
    .create();
  Logger.log('Trigger de backup diario instalado (≈03:00).');
}

// Elimina cualquier trigger existente de backupDiario (para reinstalar sin duplicar).
function eliminarTriggerBackup() {
  var triggers = ScriptApp.getProjectTriggers();
  var n = 0;
  triggers.forEach(function (t) {
    if (t.getHandlerFunction() === 'backupDiario') {
      ScriptApp.deleteTrigger(t);
      n++;
    }
  });
  if (n) Logger.log('Triggers de backup eliminados: ' + n);
}

// ── Calcula el lunes siguiente a N días desde hoy ──────
function calcularLunesSiguiente(diasExtra) {
  var hoy = new Date();
  var fecha = new Date(hoy.getFullYear(), hoy.getMonth(), hoy.getDate() + parseInt(diasExtra || 0));
  var dow = fecha.getDay(); // 0=Dom, 1=Lun, …, 6=Sab
  var diff = dow === 0 ? 1 : (dow === 1 ? 0 : 8 - dow);
  var lunes = new Date(fecha.getTime() + diff * 86400000);
  return Utilities.formatDate(lunes, 'America/Argentina/Buenos_Aires', 'yyyy-MM-dd');
}

// ── Formatea fecha ISO a DD/MM/YYYY ────────────────────
function _fmtFecha(iso) {
  if (!iso) return '-';
  var p = iso.split('-');
  return p.length === 3 ? p[2] + '/' + p[1] + '/' + p[0] : iso;
}

      + 'Devolvé ÚNICAMENTE este JSON sin markdown:\n'
      + '{\n'
      + '  "proveedor": "nombre del proveedor",\n'
      + '  "plazoTexto": "Inmediata | X dias | descripción del plazo más largo",\n'
      + '  "plazoMaxDias": número_entero,\n'
      + '  "piezasSinDisponibilidad": ["descripción pieza 1", "descripción pieza 2"]\n'
      + '}\n'
      + 'piezasSinDisponibilidad: listá CADA repuesto cuya columna "Disp." sea "SD". Si ninguno tiene SD, devolvé [].';
    var gRes = UrlFetchApp.fetch(
      'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=' + apiKey,
      { method:'post', contentType:'application/json', muteHttpExceptions:true,
        payload: JSON.stringify({ contents:[{ parts:[
          { inlineData:{ mimeType: data.mimeType, data: b64 } },
          { text: prompt }
        ]}], generationConfig:{ temperature:0.1, maxOutputTokens:2048, thinkingConfig:{ thinkingBudget:0 } } })
      }
    );
    // Validación defensiva: Gemini puede devolver error HTTP o respuesta sin candidates
    var gCode = gRes.getResponseCode();
    var gBody;
    try { gBody = JSON.parse(gRes.getContentText()); }
    catch(e) { return resp({ok: false, msg: 'Respuesta inválida de Gemini (HTTP ' + gCode + ')'}); }
    if (gCode !== 200) {
      var gErr = (gBody.error && gBody.error.message) ? gBody.error.message : 'HTTP ' + gCode;
      return resp({ok: false, msg: 'Gemini API: ' + gErr});
    }
    var gCand = gBody.candidates && gBody.candidates[0];
    var gParts = gCand && gCand.content && gCand.content.parts;
    if (!gParts || !gParts[0] || typeof gParts[0].text !== 'string') {
      var fr = gCand && gCand.finishReason ? ' (finishReason: ' + gCand.finishReason + ')' : '';
      return resp({ok: false, msg: 'Gemini no devolvió texto legible' + fr + '. Probá de nuevo o verificá que la orden sea legible.'});
    }
    var gRaw  = gParts[0].text.trim();
    // Extraer el primer bloque JSON {...} de la respuesta (Gemini a veces agrega texto extra)
    var jsonMatch = gRaw.match(/\{[\s\S]*\}/);
    if (!jsonMatch) return resp({ok: false, msg: 'Gemini no devolvió JSON válido. Respuesta: ' + gRaw.substring(0, 200)});
    // Reemplazar saltos de línea literales dentro del JSON (JSON.parse no los tolera en strings)
    var cleanJson = jsonMatch[0].replace(/[\r\n\t]/g, ' ').replace(/\s{2,}/g, ' ');
    var gData = JSON.parse(cleanJson);
    var proveedor              = gData.proveedor              || 'Sin datos';
    var plazoTexto             = gData.plazoTexto             || 'a confirmar';
    var plazoMaxDias           = parseInt(gData.plazoMaxDias) || 10;
    var piezasSinDisponibilidad = Array.isArray(gData.piezasSinDisponibilidad) ? gData.piezasSinDisponibilidad : [];
    var tieneSinDisponibilidad  = piezasSinDisponibilidad.length > 0;
    var turnoEstimado = calcularLunesSiguiente(plazoMaxDias);

    // Guardar en VerifData
    var sheet = getSheet();
    var rows  = sheet.getDataRange().getValues();
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][0] == data.id) {
        var vd = _getVD(rows, i, sheet);
        vd.ordenCompraId  = ordenId;
        vd.ordenCompraUrl = ordenUrl;
        vd.proveedor               = proveedor;
        vd.plazoTexto              = plazoTexto;
        vd.plazoMaxDias            = plazoMaxDias;
        vd.piezasSinDisponibilidad = piezasSinDisponibilidad;
        vd.tieneSinDisponibilidad  = tieneSinDisponibilidad;
        vd.turnoEstimado           = turnoEstimado;
        _setVD(sheet, i+1, vd);
        return resp({ok:true, proveedor:proveedor, plazoTexto:plazoTexto,
                     plazoMaxDias:plazoMaxDias, turnoEstimado:turnoEstimado,
                     piezasSinDisponibilidad:piezasSinDisponibilidad,
                     tieneSinDisponibilidad:tieneSinDisponibilidad,

var HONORARIOS_TARIFA = {
  'Remota':               4000,
  'Por fotos':            4000,
  'Ampliaciones de calle':8000,
  'Inspecciones Calle':   10000,
  'Posible DT':           12000
};

// Formatea número como moneda argentina: $10.000
function _fmtPesos(n) {
  return '$' + Number(n).toLocaleString('es-AR', {minimumFractionDigits:0, maximumFractionDigits:0});
}

// Devuelve YYYY-MM del mes anterior al actual
function _mesAnterior() {
  var d = new Date();
  d.setDate(1);
  d.setMonth(d.getMonth() - 1);
  var mes  = String(d.getMonth() + 1).padStart(2, '0');
  var anio = String(d.getFullYear());
  return anio + '-' + mes;
}

// Parsea "YYYY-MM" → {anio, mes} (números)
function _parseMesAnio(ym) {
  var p = String(ym || '').split('-');
  return { anio: parseInt(p[0]), mes: parseInt(p[1]) };
}

// Calcula el resumen de honorarios para un período dado.
// mesAnio: "YYYY-MM". Si no se pasa, usa el mes anterior.
// Devuelve array de {inspector, mail, inspecciones:[{...}], total}
function _calcularHonorarios(mesAnio) {
  mesAnio = mesAnio || _mesAnterior();
  var ma = _parseMesAnio(mesAnio);
  var sheet = getSheet();
  var rows  = sheet.getDataRange().getValues();
  var headers = rows[0];

  // Agrupar por inspector
  var porInspector = {};
  for (var i = 1; i < rows.length; i++) {
    var fechaInsp = String(rows[i][4] || ''); // FechaInsp
    if (!fechaInsp) continue;
    // Comparar año y mes de FechaInsp
    var d;
    try { d = new Date(fechaInsp); } catch(e) { continue; }
    if (isNaN(d.getTime())) continue;
    if (d.getFullYear() !== ma.anio || (d.getMonth() + 1) !== ma.mes) continue;

    var inspector = String(rows[i][17] || '').trim(); // Inspector
    var mailInsp  = String(rows[i][18] || '').trim(); // MailInspector
    if (!inspector) continue;

    // Leer tipoHonorario del Relevamiento JSON
    var tipoHonorario = '';
    var rawRel = rows[i][21] || ''; // Relevamiento
    if (rawRel) {
      try {
        var rel = JSON.parse(rawRel);
        tipoHonorario = rel.tipoHonorario || '';
      } catch(e) {}
    }
    var monto = tipoHonorario ? (HONORARIOS_TARIFA[tipoHonorario] || 0) : 0;

    if (!porInspector[inspector]) {
      porInspector[inspector] = { inspector: inspector, mail: mailInsp, inspecciones: [], total: 0 };
    }
    porInspector[inspector].inspecciones.push({
      siniestro:     String(rows[i][1] || ''),
      compania:      String(rows[i][2] || ''),
      patente:       String(rows[i][6] || ''),
      fechaInsp:     fechaInsp,
      tipoHonorario: tipoHonorario,
      monto:         monto
    });
    porInspector[inspector].total += monto;
  }

  return Object.keys(porInspector).map(function(k) { return porInspector[k]; });
}

// Endpoint: devuelve resumen de honorarios.
// Admin → todos los inspectores. Inspector → solo los suyos.
function getResumenHonorarios(data) {
  var mesAnio = data.mesAnio || _mesAnterior();
  var resumenes = _calcularHonorarios(mesAnio);
  if (data.usuario === 'Administrador') {
    return resp({ok: true, mesAnio: mesAnio, resumenes: resumenes});
  }
  // Inspector: filtra solo sus propias inspecciones
  var propio = resumenes.find(function(r) { return r.inspector === data.usuario; });
  if (!propio) {
    return resp({ok: true, mesAnio: mesAnio, resumenes: []});
  }
  return resp({ok: true, mesAnio: mesAnio, resumenes: [propio]});
}

function interpretarVoz(data) {
  if (!data.transcript || !data.piezas) return resp({ok: false, msg: 'transcript y piezas requeridos'});
  var apiKey = PropertiesService.getScriptProperties().getProperty('GEMINI_KEY');
  if (!apiKey) return resp({ok: false, msg: 'GEMINI_KEY no configurada'});

  var listaStr = '';
  try {
    listaStr = data.piezas.map(function(p, i) {
      return i + ': ' + p.nombre + ' [' + p.zona + ']';
    }).join('\n');
  } catch(e) {
    return resp({ok: false, msg: 'Error en lista de piezas: ' + e.toString()});
  }

  var prompt = 'Sos un asistente para perito de autos. El inspector dico lo siguiente por voz:\n'
    + '"' + data.transcript + '"\n\n'
    + 'Lista de piezas disponibles (formato: indice: nombre [zona]):\n'
    + listaStr + '\n\n'
    + 'Identificá qué piezas mencionó el inspector y qué acción corresponde a cada una. '
    + 'Las acciones posibles son: Reparar, Sustituir, Pintar, OK. '
    + 'Si mencionó roto, dañado, golpeado sin especificar = Reparar. '
    + 'Si mencionó cambiar, reemplazar, nuevo = Sustituir. '
    + 'Si mencionó rayado, pintado = Pintar. '
    + 'Para Reparar, indica nivel: leve, medio, grave. '
    + 'Respondé SOLO con JSON valido, sin texto adicional, con este formato:\n'
    + '[{"idx": 0, "pieza": "nombre", "acciones": ["Reparar"], "nivel": "medio"}]\n'
    + 'Si no reconocés piezas, respondé con array vacío: []';

  var payload = {
    contents: [{ parts: [{ text: prompt }] }],
    generationConfig: { temperature: 0 },
    thinkingConfig: { thinkingBudget: 0 }
  };

  try {
    var url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=' + apiKey;
    var res = UrlFetchApp.fetch(url, {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    });
    var json = JSON.parse(res.getContentText());
    if (!json.candidates || !json.candidates[0] || !json.candidates[0].content ||
        !json.candidates[0].content.parts || !json.candidates[0].content.parts[0]) {
      return resp({ok: false, msg: 'Respuesta vacia de Gemini'});
    }
    var texto = json.candidates[0].content.parts[0].text || '';
    texto = texto
      .replace(new RegExp('^```json\\s*', 'i'), '')
      .replace(new RegExp('^```\\s*', 'i'), '')
      .replace(new RegExp('```\\s*$'), '')
      .trim();
    var resultado = JSON.parse(texto);
    if (!Array.isArray(resultado)) return resp({ok: false, msg: 'Formato inesperado de Gemini'});
    return resp({ok: true, resultado: resultado});
  } catch(e) {
    return resp({ok: false, msg: 'Error interpretarVoz: ' + e.toString()});
  }
}
