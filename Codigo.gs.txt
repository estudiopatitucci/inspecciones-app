function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var action = data.action;
    // relevamientoTaller usa tallerToken propio, no SECRET_TOKEN
    if (action === 'relevamientoTaller') return recibirRelTaller(data);
    if (action === 'uploadFotoTaller') return uploadFotoTaller(data);
    // El formulario de contacto público no requiere token
    if (action !== 'contacto' && data.token !== SECRET_TOKEN) {
      return resp({ok: false, msg: 'No autorizado'});
    }
    if (action === 'guardar') return guardarInspeccion(data);
    if (action === 'actualizar') return actualizarInspeccion(data);
    if (action === 'editar') return actualizarInspeccion(data);
    if (action === 'eliminar') return eliminarInspeccion(data);
    if (action === 'listar') return listarInspecciones();
    if (action === 'recordatorio') return enviarRecordatorio(data);
    if (action === 'archivar') return archivarInspecciones();
    if (action === 'contacto') return enviarContacto(data);
    if (action === 'generarTokenTaller') return generarTokenTaller(data);
    if (action === 'uploadFoto') return uploadFotoSingle(data);
    return resp({ok:false, msg:'Accion desconocida'});
  } catch(err) {
    return resp({ok:false, msg:err.toString()});
  }
}

function doGet(e) {

}

// Devuelve el número de columna (1-based) de DanosDen, creándola si no existe.
function getDanosDenCol(sheet) {
  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  var idx = headers.indexOf('DanosDen');
  if (idx !== -1) return idx + 1;
  // No existe → agregar al final
  var col = sheet.getLastColumn() + 1;
  sheet.getRange(1, col).setValue('DanosDen');
  return col;
}

function leerInspecciones(sheet) {
  var rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) return [];
  var headers = rows[0];
  var result = [];
  for (var i = 1; i < rows.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = rows[i][j];
    }
    if (obj.Relevamiento) {
      try { obj.Relevamiento = JSON.parse(obj.Relevamiento); }
      catch(e) { obj.Relevamiento = null; }
    }
    if (obj.Devoluciones) {
      try { obj.Devoluciones = JSON.parse(obj.Devoluciones); }
      catch(e) { obj.Devoluciones = []; }
    }
    result.push(obj);
  }
  return result;
}

function guardarInspeccion(data) {
  var sheet = getSheet();
  var id = Date.now().toString();
  sheet.appendRow([
    id, data.sin, data.cia, data.tipo, data.finsp, data.fsin,
    data.pat, data.mod, data.anio, data.chasis, data.motor, data.km,
    data.aseg, data.tel, data.dir, data.localidad, data.danos,
    data.inspector, data.mail, data.estado || 'Asignada',
    data.mailEnviado ? 'Si' : 'No',
    '', '', new Date().toISOString()
  ]);
  // Guardar DanosDen (columna auto-detectada / creada)
  if (data.danosDen) {
    var lastRow = sheet.getLastRow();
    var ddCol = getDanosDenCol(sheet);
    sheet.getRange(lastRow, ddCol).setValue(data.danosDen);
  }
  if (data.enviarMail && data.mail) {
    enviarMail(data);
  }
  // Si se asigna directo al taller, generar token inmediatamente
  if (data.asignarTaller) {
    var token = generarToken32();
    var exp = new Date();
    exp.setHours(exp.getHours() + 48);
    var lastRow = sheet.getLastRow();
    sheet.getRange(lastRow, 25).setValue(token);
    sheet.getRange(lastRow, 26).setValue(exp.toISOString());
    var appUrl = 'https://estudiopatitucci.github.io/inspecciones-app/?taller=' + token;
    return resp({ok:true, id:id, link:appUrl, exp:exp.toISOString()});
  }
  return resp({ok:true, id:id});
}

// Nombre de la carpeta madre donde se guardan todas las inspecciones
const CARPETA_MADRE = 'Inspecciones Patitucci - Fotos';

function getCarpetaMadre() {
  var folders = DriveApp.getFoldersByName(CARPETA_MADRE);
  if (folders.hasNext()) return folders.next();
  // Si no existe, la crea y comparte con los viewers autorizados
  var nueva = DriveApp.createFolder(CARPETA_MADRE);
  try { compartirConAutorizados(nueva); } catch(e) {}
  return nueva;
}

function getOrCreateFolder(name) {
  var madre = getCarpetaMadre();
  // Buscar la subcarpeta dentro de la carpeta madre
  var subs = madre.getFoldersByName(name);
  if (subs.hasNext()) return subs.next();
  // Si no existe dentro de la madre, buscar en el raíz (carpetas viejas sueltas) y moverlas
  var suelta = DriveApp.getFoldersByName(name);
  if (suelta.hasNext()) {
    var carpetaSuelta = suelta.next();
    madre.addFolder(carpetaSuelta);
    DriveApp.getRootFolder().removeFolder(carpetaSuelta);
    return carpetaSuelta;
  }
  // Crear dentro de la carpeta madre
  return madre.createFolder(name);
}

function archivarInspecciones() {
  var sheet = getSheet();
  var archivoSheet = getArchivoSheet();
  var rows = sheet.getDataRange().getValues();
  var headers = rows[0];
  var hoy = new Date();
  var seisM = new Date(hoy);
  seisM.setMonth(seisM.getMonth() - 6);

  var archivadas = 0;
  // Recorrer de abajo para arriba para no alterar índices al borrar
  for (var i = rows.length - 1; i >= 1; i--) {
    var estado = rows[i][19]; // columna Estado (índice 19)
    var fechaInsp = rows[i][4]; // columna FechaInsp (índice 4)
    if (estado !== 'Cerradas') continue;
    var fecha = new Date(fechaInsp);
    if (isNaN(fecha.getTime())) continue;
    if (fecha > seisM) continue;
    // Mover a Archivo
    var fila = rows[i].slice();
    fila.push(new Date().toISOString()); // ArchivedAt
    archivoSheet.appendRow(fila);
    sheet.deleteRow(i + 1);
    archivadas++;
  }
  return {ok: true, archivadas: archivadas};
}

function migrarPermisosExistentes() {
  var resultados = { ok: 0, err: 0, carpetas: 0 };
  
  // Buscar todas las carpetas que empiecen con "Inspeccion_"
  var carpetasIter = DriveApp.getFoldersByName('');
  
  // Buscar en el Drive del usuario
  var query = DriveApp.searchFolders('title contains "Inspeccion_"');
  
  while (query.hasNext()) {
    var carpeta = query.next();
    resultados.carpetas++;
    Logger.log('Procesando carpeta: ' + carpeta.getName());
    
    var archivos = carpeta.getFiles();
    while (archivos.hasNext()) {
      var archivo = archivos.next();
      try {
        archivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
        resultados.ok++;
        Logger.log('  ✓ ' + archivo.getName() + ' — permiso aplicado');
      } catch(e) {
        resultados.err++;
        Logger.log('  ✕ ' + archivo.getName() + ' — ERROR: ' + e.toString());
      }
    }
  }

  Logger.log('');
  Logger.log('════ Migración completada ════');
  Logger.log('Carpetas procesadas: ' + resultados.carpetas);
  Logger.log('Archivos OK: ' + resultados.ok);
  Logger.log('Errores: ' + resultados.err);
}

//  MIGRACIÓN — mover carpetas sueltas a la carpeta madre
//  Correr UNA SOLA VEZ después de hacer deploy
//  Pasos: seleccionar "migrarCarpetasAMadre" → Ejecutar → Ver Registros
// ══════════════════════════════════════════════════════
function migrarCarpetasAMadre() {
  var madre = getCarpetaMadre();
  var query = DriveApp.searchFolders('title contains "Inspeccion_"');
  var movidas = 0, ya = 0;
  while (query.hasNext()) {
    var carpeta = query.next();
    var parents = carpeta.getParents();
    var estaEnMadre = false;
    while (parents.hasNext()) {
      if (parents.next().getId() === madre.getId()) { estaEnMadre = true; break; }
    }
    if (estaEnMadre) { ya++; continue; }
    madre.addFolder(carpeta);
    DriveApp.getRootFolder().removeFolder(carpeta);
    movidas++;
    Logger.log('Movida: ' + carpeta.getName());
  }
  Logger.log('Ya estaban en madre: ' + ya + ' | Movidas ahora: ' + movidas);
}

// ================================================================
//  FEATURE: LINK PARA TALLERES
//  El admin genera un link unico con vencimiento de 48 hs.
//  El taller lo abre, carga el relevamiento y lo envia.
//  No requiere login. El token se invalida al usarse o al vencer.
// ================================================================

function generarToken32() {
  var chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  var t = '';
  for (var i = 0; i < 32; i++) t += chars[Math.floor(Math.random() * chars.length)];
  return t;
}

function generarTokenTaller(data) {
  if (!data.id) return resp({ok: false, msg: 'ID requerido'});
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.id) {
      var token = generarToken32();
      var exp = new Date();
      exp.setHours(exp.getHours() + 48);
      sheet.getRange(i+1, 25).setValue(token);        // col TallerToken
      sheet.getRange(i+1, 26).setValue(exp.toISOString()); // col TallerTokenExp
      sheet.getRange(i+1, 20).setValue('Esperando taller'); // col Estado
      var appUrl = 'https://estudiopatitucci.github.io/inspecciones-app/?taller=' + token;
      return resp({ok: true, link: appUrl, exp: exp.toISOString()});
    }
  }
  return resp({ok: false, msg: 'Inspeccion no encontrada'});
}

  var headers = rows[0];
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][24] === tallerToken) {               // col 25 = index 24
      var exp = rows[i][25];                          // col 26 = index 25
      if (exp && new Date() > new Date(exp)) {
        return resp({ok: false, msg: 'Link vencido. Solicita uno nuevo al estudio.'});
      }
      if (rows[i][21]) {                              // col Relevamiento = index 21
        return resp({ok: false, msg: 'Este link ya fue utilizado. El relevamiento ya fue enviado.'});
      }
      return resp({ok: true, inspeccion: {
        id:         rows[i][0],
        siniestro:  rows[i][1],
        compania:   rows[i][2],
        tipo:       rows[i][3],
        fechaInsp:  rows[i][4],
        patente:    rows[i][6],
        modelo:     rows[i][7],
        anio:       rows[i][8],
        asegurado:  rows[i][12],
        tel:        rows[i][13],
        direccion:  rows[i][14],
        localidad:  rows[i][15],
        danos:      rows[i][16]
      }});
    }
  }
  return resp({ok: false, msg: 'Link invalido o expirado.'});
}

function recibirRelTaller(data) {
  var tallerToken = data.tallerToken;
  if (!tallerToken) return resp({ok: false, msg: 'Token requerido'});
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][24] === tallerToken) {
      var exp = rows[i][25];
      if (exp && new Date() > new Date(exp)) {
        return resp({ok: false, msg: 'Link vencido'});
      }
      if (rows[i][21]) {
        return resp({ok: false, msg: 'Este relevamiento ya fue enviado anteriormente.'});
      }
      var rel = data.relevamiento || {};
      // Subir fotos a Drive (misma logica que actualizarInspeccion)
      // Flujo nuevo: fotos ya subidas progresivamente
      if (rel.fotosLinks && rel.fotosLinks.length > 0 && (!rel.fotos || rel.fotos.length === 0)) {
        rel.cantFotos = rel.fotosLinks.length;
        try {
          var idInsp = rows[i][0];
          var folder = getOrCreateFolder('Inspeccion_' + idInsp);
          rel.carpetaUrl = folder.getUrl();
        } catch(eF) {}
      }
      // Flujo legacy: base64
      if (rel.fotos && rel.fotos.length > 0) {
        try {
          var id = rows[i][0];
          var folder = getOrCreateFolder('Inspeccion_' + id);
          try { compartirConAutorizados(folder); } catch(eF) {}
          var links = [];
          rel.fotos.forEach(function(base64, idx) {
            try {
              var b64 = base64.replace(/^data:image\/\w+;base64,/, '');
              var blob = Utilities.newBlob(Utilities.base64Decode(b64), 'image/jpeg', 'foto_' + (idx+1) + '.jpg');
              var file = folder.createFile(blob);
              try { file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(eS) {}
              try { compartirConAutorizados(file); } catch(eV) {}
              var fileId = file.getId();
              links.push({viewUrl: file.getUrl(), id: fileId, thumbUrl: 'https://lh3.googleusercontent.com/d/' + fileId + '=w400'});
            } catch(e) {}
          });
          rel.fotosLinks = links;
          rel.cantFotos = links.length;
          rel.carpetaUrl = folder.getUrl();
        } catch(e) {}
        delete rel.fotos;
      }
      rel.origenTaller = true;
      sheet.getRange(i+1, 22).setValue(JSON.stringify(rel)); // Relevamiento
      sheet.getRange(i+1, 20).setValue('Pendiente de verificar'); // Estado
      // Invalidar token para que no se pueda usar dos veces
      sheet.getRange(i+1, 25).setValue('USADO');
      sheet.getRange(i+1, 26).setValue('');
      // Notificar al administrador
      try {
        var siniestro = rows[i][1] || '-';
        var patente   = rows[i][6] || '-';
        var modelo    = rows[i][7] || '-';
        var tallerNombre = rel.taller || 'Sin nombre';
        MailApp.sendEmail({
          to: 'patitucciestudio@gmail.com',
          subject: 'Relevamiento de taller recibido — Siniestro ' + siniestro,
          body: 'ESTUDIO PATITUCCI\n'
            + 'Pericias - Daños Materiales - Homologación CESVI\n\n'
            + 'El taller cargó el relevamiento de la siguiente inspección:\n\n'
            + 'N° Siniestro: ' + siniestro + '\n'
            + 'Patente: ' + patente + '\n'
            + 'Modelo: ' + modelo + '\n'
            + 'Taller: ' + tallerNombre + '\n\n'
            + 'La inspección fue movida a "Pendiente de verificar".\n'
            + 'Ingresá al panel para revisarla.\n\n'
            + 'Saludos,\nEstudio Patitucci'
        });
      } catch(eM) {}
      return resp({ok: true});
    }
  }
  return resp({ok: false, msg: 'Token invalido'});
}

// ════════════════════════════════════════════════════════
//  UPLOAD PROGRESIVO DE FOTOS (nueva arquitectura)
//  Cada foto se sube individualmente al momento de seleccionarla,
//  para que el envío final solo transmita metadatos y links.
// ════════════════════════════════════════════════════════

function uploadFotoSingle(data) {
  if (!data.id) return resp({ok: false, msg: 'ID requerido'});
  try {
    var folder = getOrCreateFolder('Inspeccion_' + data.id);
    try { compartirConAutorizados(folder); } catch(e) {}
    var b64 = data.foto.replace(/^data:image\/\w+;base64,/, '');
    var idx = (data.idx !== undefined) ? data.idx + 1 : Date.now();
    var blob = Utilities.newBlob(Utilities.base64Decode(b64), 'image/jpeg', 'foto_' + idx + '.jpg');
    var file = folder.createFile(blob);
    try { file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(e) {}
    try { compartirConAutorizados(file); } catch(e) {}
    var fileId = file.getId();
    return resp({ok: true, link: {
      viewUrl: file.getUrl(),
      id: fileId,
      thumbUrl: 'https://lh3.googleusercontent.com/d/' + fileId + '=w400'
    }});
  } catch(err) {
    return resp({ok: false, msg: err.toString()});
  }
}

function uploadFotoTaller(data) {
  var token = data.tallerToken;
  if (!token) return resp({ok: false, msg: 'Token requerido'});
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][24] === token) {
      var exp = rows[i][25];
      if (exp && new Date() > new Date(exp)) return resp({ok: false, msg: 'Token vencido'});
      data.id = rows[i][0];
      return uploadFotoSingle(data);
    }
  }
  return resp({ok: false, msg: 'Token inválido'});
}
