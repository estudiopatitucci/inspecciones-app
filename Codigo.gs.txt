const SHEET_ID = '1_socHaF14orZbtqiMxI5SKYMWEEGShw4gY_SEtMNN0w';
const SHEET_NAME = 'Inspecciones';
const ARCHIVO_NAME = 'Archivo';

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var action = data.action;
    if (action === 'guardar') return guardarInspeccion(data);
    if (action === 'actualizar') return actualizarInspeccion(data);
    if (action === 'eliminar') return eliminarInspeccion(data);
    if (action === 'listar') return listarInspecciones();
    if (action === 'recordatorio') return enviarRecordatorio(data);
    if (action === 'archivar') return archivarInspecciones();
    if (action === 'contacto') return enviarContacto(data);
    return resp({ok:false, msg:'Accion desconocida'});
  } catch(err) {
    return resp({ok:false, msg:err.toString()});
  }
}

function doGet(e) {
  var action = e && e.parameter && e.parameter.action;
  if (action === 'getArchivadas') return getArchivadas();
  return listarInspecciones();
}

function getSheet() {
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(['ID','Siniestro','Compania','Tipo','FechaInsp','FechaSin','Patente','Modelo','Anio','Chasis','Motor','Km','Asegurado','Tel','Direccion','Localidad','Danos','Inspector','MailInspector','Estado','MailEnviado','Relevamiento','Devoluciones','CreadaEn']);
  }
  return sheet;
}

function getArchivoSheet() {
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var sheet = ss.getSheetByName(ARCHIVO_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(ARCHIVO_NAME);
    sheet.appendRow(['ID','Siniestro','Compania','Tipo','FechaInsp','FechaSin','Patente','Modelo','Anio','Chasis','Motor','Km','Asegurado','Tel','Direccion','Localidad','Danos','Inspector','MailInspector','Estado','MailEnviado','Relevamiento','Devoluciones','CreadaEn','ArchivedAt']);
  }
  return sheet;
}

function leerInspecciones(sheet) {
  var rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) return [];
  var headers = rows[0];
  var result = [];
  for (var i = 1; i < rows.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = rows[i][j];
    }
    if (obj.Relevamiento) {
      try { obj.Relevamiento = JSON.parse(obj.Relevamiento); }
      catch(e) { obj.Relevamiento = null; }
    }
    if (obj.Devoluciones) {
      try { obj.Devoluciones = JSON.parse(obj.Devoluciones); }
      catch(e) { obj.Devoluciones = []; }
    }
    result.push(obj);
  }
  return result;
}

function guardarInspeccion(data) {
  var sheet = getSheet();
  var id = Date.now().toString();
  sheet.appendRow([
    id, data.sin, data.cia, data.tipo, data.finsp, data.fsin,
    data.pat, data.mod, data.anio, data.chasis, data.motor, data.km,
    data.aseg, data.tel, data.dir, data.localidad, data.danos,
    data.inspector, data.mail, data.estado || 'Asignada',
    data.mailEnviado ? 'Si' : 'No',
    '', '', new Date().toISOString()
  ]);
  if (data.enviarMail && data.mail) {
    enviarMail(data);
  }
  return resp({ok:true, id:id});
}

function getOrCreateFolder(name) {
  var folders = DriveApp.getFoldersByName(name);
  if (folders.hasNext()) return folders.next();
  return DriveApp.createFolder(name);
}

function actualizarInspeccion(data) {
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.id) {
      if (data.estado !== undefined) sheet.getRange(i+1, 20).setValue(data.estado);
      if (data.relevamiento !== undefined) {
        var rel = data.relevamiento;
        // Si hay fotos en base64, subirlas a Drive
        if (rel.fotos && rel.fotos.length > 0) {
          try {
            var folder = getOrCreateFolder('Inspeccion_' + data.id);
            var links = [];
            rel.fotos.forEach(function(base64, idx) {
              try {
                var b64 = base64.replace(/^data:image\/\w+;base64,/, '');
                var blob = Utilities.newBlob(Utilities.base64Decode(b64), 'image/jpeg', 'foto_' + (idx+1) + '.jpg');
                var file = folder.createFile(blob);
                file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
                links.push({viewUrl: file.getUrl(), id: file.getId()});
              } catch(e) {}
            });
            rel.fotosLinks = links;
            rel.cantFotos = links.length;
            rel.carpetaUrl = folder.getUrl();
          } catch(e) {}
          delete rel.fotos; // Eliminar base64 antes de guardar en Sheets
        }
        sheet.getRange(i+1, 22).setValue(JSON.stringify(rel));
        // Devolver links al frontend
        if (rel.fotosLinks) {
          return resp({ok: true, relevamiento: {fotosLinks: rel.fotosLinks, cantFotos: rel.cantFotos, carpetaUrl: rel.carpetaUrl}});
        }
      }
      if (data.devoluciones !== undefined) sheet.getRange(i+1, 23).setValue(JSON.stringify(data.devoluciones));
      if (data.mailEnviado !== undefined) sheet.getRange(i+1, 21).setValue(data.mailEnviado ? 'Si' : 'No');
      if (data.inspector !== undefined) {
        sheet.getRange(i+1, 18).setValue(data.inspector);
        if (data.mail !== undefined) sheet.getRange(i+1, 19).setValue(data.mail);
        if (data.notificar && data.mail) {
          enviarMail({
            inspector: data.inspector,
            mail: data.mail,
            sin: rows[i][1], cia: rows[i][2], tipo: rows[i][3],
            finsp: rows[i][4], fsin: rows[i][5], pat: rows[i][6],
            mod: rows[i][7], anio: rows[i][8], chasis: rows[i][9],
            motor: rows[i][10], km: rows[i][11], aseg: rows[i][12],
            tel: rows[i][13], dir: rows[i][14], localidad: rows[i][15],
            danos: rows[i][16]
          });
        }
      }
      return resp({ok:true});
    }
  }
  return resp({ok:false, msg:'No encontrado'});
}

function eliminarInspeccion(data) {
  var sheet = getSheet();
  var rows = sheet.getDataRange().getValues();
  for (var i = 1; i < rows.length; i++) {
    if (rows[i][0] == data.id) {
      sheet.deleteRow(i+1);
      return resp({ok:true});
    }
  }
  return resp({ok:false, msg:'No encontrado'});
}

function listarInspecciones() {
  return resp({ok:true, data: leerInspecciones(getSheet())});
}

function getArchivadas() {
  return resp({ok:true, data: leerInspecciones(getArchivoSheet())});
}

function archivarInspecciones() {
  var sheet = getSheet();
  var archivoSheet = getArchivoSheet();
  var rows = sheet.getDataRange().getValues();
  var headers = rows[0];
  var hoy = new Date();
  var seisM = new Date(hoy);
  seisM.setMonth(seisM.getMonth() - 6);

  var archivadas = 0;
  // Recorrer de abajo para arriba para no alterar índices al borrar
  for (var i = rows.length - 1; i >= 1; i--) {
    var estado = rows[i][19]; // columna Estado (índice 19)
    var fechaInsp = rows[i][4]; // columna FechaInsp (índice 4)
    if (estado !== 'Cerradas') continue;
    var fecha = new Date(fechaInsp);
    if (isNaN(fecha.getTime())) continue;
    if (fecha > seisM) continue;
    // Mover a Archivo
    var fila = rows[i].slice();
    fila.push(new Date().toISOString()); // ArchivedAt
    archivoSheet.appendRow(fila);
    sheet.deleteRow(i + 1);
    archivadas++;
  }
  return {ok: true, archivadas: archivadas};
}
